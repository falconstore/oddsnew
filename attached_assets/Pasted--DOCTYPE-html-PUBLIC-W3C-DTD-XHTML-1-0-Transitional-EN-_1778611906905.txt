<script>
	function listEarningsBySystem(button, startDate, endDate){
		var sub_rows_selector = startDate == endDate ? '.item-' + startDate : '.item-' + startDate.substring(0, 7);
		if($(button).data('status') == 'opening'){
			return;
		}
		if($(button).data('status') == 'closed'){
            $(button).data('status', 'opening');
			$(button).css('color', 'transparent');
			var loader = addLoader($(button), 0, 12);
            
            $.ajax({
                url: 'earnings/aListEarningsBySystem',
                type: 'GET',
                dataType: 'json',
                data: {
					affiliate_id: 6,
                    startDate: startDate,
					endDate: endDate,
                    currency: PAGINATION.filters.currency,
					affiliate_cookie: PAGINATION.filters.affiliate_cookie ? PAGINATION.filters.affiliate_cookie : null,
					affiliate_earnings_types: PAGINATION.filters.affiliate_earnings_types
                },
                success: function(response){
                    if(response.status == 'not-logged'){
                        window.location.href = 'index.php';
                        return;
                    }
					else if(response.status == 'error'){
						ALERT.error(response.message);
						return;
					}
					else if(response.status == 'success'){
						$(button).data('status', 'open');
						$(button).closest('tr').after(response.html);
					}
					removeLoader(loader);
					$(button).css('color', 'inherit');
					$(button).text('keyboard_arrow_up');
                }
            });
        }
        else if($(button).data('status') == 'closed-with-data'){
            $(button).data('status', 'open');
            $(button).text('keyboard_arrow_up');
            $(button).closest('tr').siblings('tr' + sub_rows_selector).show();
        }
        else {
            $(button).data('status', 'closed-with-data');
            $(button).text('keyboard_arrow_down');
            $(button).closest('tr').siblings('tr' + sub_rows_selector).hide();
        }
	}
	
	function reload(){
		location.reload();
	}
